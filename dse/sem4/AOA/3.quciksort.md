void  quicksort(int a[],int LB, int UB)
{
int  LOC;
if(LB<UB)
{
LOC=partition(a,LB,UB);
quicksort(a,LB,LOC-1);
quicksort(a,LOC+1,UB);
}
}
int  partition(int a[],int LB,int UB)
{
int  START, PIVOT, END, temp;
PIVOT=a[LB];
START=LB;
END=UB;
while(START<END)
{
while(a[START]<=PIVOT)
{
START++;
}
while(a[END]>PIVOT)
{
END--;
}
if(START<END)
{
temp=a[START];
a[START]=a[END];
a[END]=temp;
}
}
temp=a[LB];
a[LB]=a[END];
a[END]=temp;
return END;
}
